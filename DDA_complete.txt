create table department (
  dept_id varchar(3) not null,
  department_name varchar(40) not null,
  department_duty varchar(50) not null,
primary key(dept_id),
constraint chk_dept_id check (REGEXP_LIKE(dept_id,'[D]{1}[0-9]{2}'))
);


create table employee(
	employee_ID varchar(10) not null,
	manager_ID varchar(10),
	employee_Name varchar(27) not null,
	employee_Gender varchar(10) not null,
	dept_ID varchar(10) not null,
	job_Position varchar(45) not null,
	total_Work_Year number(2), 
	employee_Salary number(6,2),
PRIMARY KEY (employee_ID),
CONSTRAINT chk_gender CHECK (UPPER(employee_Gender) in ('M', 'F')),
FOREIGN KEY (dept_ID) references department(dept_ID),
constraint chk_employee_ID CHECK (REGEXP_LIKE(employee_ID,'[S]{1}[T]{1}[F]{1}[F]{1}[0-9]{3}'))
);


ALTER SESSION SET NLS_DATE_FORMAT = 'DD/MM/YYYY'; 

create table test (
	test_ID           varchar(10)  not null,
	employee_ID       varchar(10)  not null,
	created_Date DATE,
	using_Date   DATE,
PRIMARY KEY (test_ID),
constraint chk_test_id CHECK (REGEXP_LIKE(test_ID ,'[T]{1}[A-Z]{1}[0-9]{4}')),
FOREIGN KEY (employee_ID ) references employee(employee_ID )
);

create table agent(
	agent_id varchar(6) not null,
	agent_name varchar(30) not null,
	agent_position varchar(30) not null,
	no_agent_team number(4) not null,
	advertisement_exp number(2),
primary key(agent_id),
CONSTRAINT chk_agent_id CHECK (REGEXP_LIKE(agent_id,'[A]{1}[G]{1}[0-9]{4}'))
);


create table recruitment_job_details(
	job_detail_id    varchar(6) not null,
	employee_ID      varchar(10) not null,
	job_department   varchar(3),
	job_position     varchar(35) not null,
	request_date     date,
	job_required_exp number(2) not null,
	job_required_edu varchar(70) not null,
primary key(job_detail_id),
CONSTRAINT chk_job_detail_id CHECK (REGEXP_LIKE(job_detail_id,'[J]{1}[D]{1}[0-9]{4}')),
foreign key(employee_ID) references employee(employee_ID)
);



create table advertisement_details(
    job_detail_id varchar(6) not null,
    agent_id varchar(6) not null,
    date_start date,
    advertisement_name varchar(60) not null,
    recruitment_platform varchar(20),
primary key(job_detail_id,agent_id),
foreign key(job_detail_id) references recruitment_job_details(job_detail_id),
foreign key(agent_id) references agent(agent_id)
);

create table CANDIDATES (
  Candidates_id varchar(5) not null,
  Candidates_name varchar(30) not null,
  Candidates_gender varchar(1) not null,
  Candidates_job varchar(35) not null,
  Candidates_expected_Salary number(10,2) not null,
  job_detail_id varchar(6) not null,
primary key(Candidates_id),
constraint chk_Candidates_id check (REGEXP_LIKE(Candidates_id,'[C]{1}[0-9]{4}')),
foreign key(job_detail_id)references recruitment_job_details(job_detail_id)
);


create table offer_letter(
	offer_letter_id		char(5)				not null,
	employee_id		varchar(10)				not null,
	new_emp_name		varchar(15)				not null,
	work_start_date		date default '01-JAN-2023',
	work_location		varchar(20)				not null,
primary key(offer_letter_id),
foreign key(employee_id) references employee(employee_id),
constraint chk_offer_letter_id check(REGEXP_LIKE(offer_letter_id,'[L]{1}[0-9]{3}'))
);


create table interview( 
	interview_id		char(5)				not null,
	interview_name		varchar(25)				not null,
	interview_date		date default '01-JAN-2023',
	candidates_attendance	varchar(7),
	interview_position	varchar(15)	,
	candidates_id		varchar(5)				not null,
	test_id			varchar(10)				not null,
primary key(interview_id),
foreign key(candidates_id) references candidates(candidates_id),
foreign key(test_id) references test(test_id),
constraint chk_interview_id check (REGEXP_LIKE(interview_id,'[I]{1}[0-9]{4}'))
);


create table RESUME( 
Resume_id  varchar(7) not null,
Candidates_ID varchar(5) not null,
Resume_date date,
Resume_job varchar(40), 
Resume_educational varchar(60),
Job_Experience number(2),
primary key(Resume_ID),
foreign key(candidates_id) references candidates(candidates_id),
constraint chk_Resume_ID check (REGEXP_LIKE(Resume_ID,'[R]{1}_[A]{1}[0-9]{4}'))
);


CREATE TABLE RESUME_SELECTION(
Resume_selection_ID varchar(8) not null,
Employee_ID varchar(10) not null,
Candidates_ID varchar(5) not null,
Resume_ID varchar(10) not null,
Selection_Description varchar(50) DEFAULT 'Fulfill all the needed requirement',
Selection_Date date,
Candidates_TestResult varchar(6),
Candidates_InterviewResult varchar(6),
primary key(Resume_selection_ID),
foreign key(Employee_ID) references EMPLOYEE(Employee_ID),
foreign key(Candidates_ID) references CANDIDATES(Candidates_ID),
foreign key(Resume_ID) references RESUME(Resume_ID),
constraint chk_Resume_selection_ID check (REGEXP_LIKE(Resume_selection_ID,'[R]{1}[E]{1}_[A-Z]{1}[0-9]{4}'))
);
